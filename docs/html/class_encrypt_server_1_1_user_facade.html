◆ find()

EncryptServer\UserFacade::find ( array  $filter,
array  $options = [] 
)

Find user by filter. 

 for filter allows keys ["email", "login", "firstname", "lastname","nickname", "blocked"]
 where
 email - user email (string)
 login - user login (string)
 firstname - user firstname (string)
 "lastname" - user lastname (string)
 "nickname" - user nickname (string)
 "blocked" - user blocked or no (boolean)
 
 for options allows keys ["count", "page", "sort"]
 where
 "count" - count records (int)
 "page" - number page (int)
 "sort" - asc|desc (string)
Parameters
array$filter
array$optionsallows keys ["count", "page", "sort"]
Returns
Response\Response
Example:
$facade = new \EncryptServer\UserFacade();
$response = $facade->find(["email" => $email, "login" => $login], ["count" => 1]);
$data = $response->getResponse();
$searchResult = $data["searchResult"];
$ar = [
"total" => $searchResult->getTotal(),
"page" => $searchResult->getPage(),
"data" => []
];
while ($searchResult->valid())
{
$ar["data"][] = $searchResult->current();
$searchResult->next();
}
